This directory contains the following programmatically generated files:

- `arc.bin`, the round constants for Poseidon's ARC layer
- `g1.bin`, a KZG SRS containing 65536 $G_1$ points
- `g2.bin`, a KZG SRS containing 65536 $G_2$ points
- `mds.bin`, the MDS matrix for Poseidon's MDS layer

`g1.bin` and `g2.bin` have been generated by the [`generate_params`][generate_params] script.

`arc.bin` and `mds.bin` have been generated by the following script:

```rs
use blstrs::Scalar;
use std::fs::File;
use std::io::{Result, Write};

const R: usize = 3;
const C: usize = 1;
const T: usize = R + C;
const NUM_FULL_ROUNDS_START: usize = 4;
const NUM_FULL_ROUNDS_END: usize = 4;
const NUM_FULL_ROUNDS: usize = NUM_FULL_ROUNDS_START + NUM_FULL_ROUNDS_END;
const NUM_PARTIAL_ROUNDS: usize = 56;
const NUM_ROUNDS: usize = NUM_FULL_ROUNDS + NUM_PARTIAL_ROUNDS;

// The round constants and MDS matrix have been generated by the official script at
// https://extgit.iaik.tugraz.at/krypto/hadeshash/-/blob/master/code/generate_parameters_grain.sage.
//
// The command line used was:
//
//   sage generate_parameters_grain.sage 1 0 255 4 8 56 0x73eda753299d7d483339d80809a1d80553bda402fffe5bfeffffffff00000001

fn make_constants() -> [Scalar; NUM_ROUNDS * T] {
    [
        // ... round constants ...
    ]
}

fn make_mds() -> [[Scalar; T]; T] {
    [
        // ... MDS matrix ...
    ]
}

fn write_constants() -> Result<()> {
    let mut bytes = vec![0u8; 32 * NUM_ROUNDS * T];
    let mut offset = 0;
    for c in make_constants() {
        bytes[offset..(offset + 32)].copy_from_slice(&c.to_bytes_le());
        offset += 32;
    }
    let mut file = File::create("arc.bin")?;
    file.write_all(bytes.as_slice())?;
    Ok(())
}

fn write_mds() -> Result<()> {
    let mut bytes = vec![0u8; 32 * NUM_ROUNDS * T];
    let mut offset = 0;
    let mds = make_mds();
    for i in 0..T {
        for j in 0..T {
            let x = mds[i][j];
            bytes[offset..(offset + 32)].copy_from_slice(&x.to_bytes_le());
            offset += 32;
        }
    }
    let mut file = File::create("mds.bin")?;
    file.write_all(bytes.as_slice())?;
    Ok(())
}

fn main() -> Result<()> {
    write_constants()?;
    write_mds()?;
    Ok(())
}
```
